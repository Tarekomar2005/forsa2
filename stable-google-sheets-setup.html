  • كل جهاز يشوف طلباته المحلية بس
  • Google Sheets محتاج re-deploy كل مرة

الخطوة 2: الكود النهائي والثابت

انسخ هذا الكود مرة واحدة فقط - لن تحتاج تغييره أبداً:

function doPost(e) { try { const data = JSON.parse(e.postData.contents); const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); // Save complete orders (الطلبات الكاملة) if (data.action === 'saveCompleteOrder') { return saveCompleteOrder(data, spreadsheet); } // Save contact forms (رسائل التواصل) if (data.action === 'saveContact') { return saveContact(data, spreadsheet); } // Save quick orders (الطلبات السريعة) if (data.action === 'saveOrder') { return saveQuickOrder(data, spreadsheet); } return createResponse('error', 'Unknown action: ' + data.action); } catch (error) { console.error('Error in doPost:', error); return createResponse('error', error.toString()); } } function doGet(e) { try { const action = e.parameter.action; const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'getCompleteOrders') { return getCompleteOrders(spreadsheet); } if (action === 'getAllData') { return getAllData(spreadsheet); } if (action === 'test') { return createResponse('success', 'Google Apps Script is working!'); } return createResponse('error', 'Unknown action: ' + action); } catch (error) { console.error('Error in doGet:', error); return createResponse('error', error.toString()); } } function saveCompleteOrder(data, spreadsheet) { let sheet = getOrCreateSheet(spreadsheet, 'Complete Orders', [ 'Timestamp', 'Order ID', 'Customer Name', 'Customer Phone', 'Customer Email', 'Customer City', 'Customer Address', 'Payment Method', 'Customer Notes', 'Products JSON', 'Total Items', 'Total Amount', 'Source', 'Device Info' ]); const timestamp = new Date().toLocaleString('ar-EG'); const products = typeof data.products === 'string' ? data.products : JSON.stringify(data.products || []); const deviceInfo = data.userAgent || 'Unknown Device'; sheet.appendRow([ timestamp, data.orderId || 'ORD-' + Date.now(), data.customerName || '', data.customerPhone || '', data.customerEmail || '', data.customerCity || '', data.customerAddress || '', data.paymentMethod || '', data.customerNotes || '', products, data.totalItems || 0, data.totalAmount || 0, 'google_sheets', deviceInfo ]); return createResponse('success', 'Order saved successfully'); } function saveContact(data, spreadsheet) { let sheet = getOrCreateSheet(spreadsheet, 'Contacts', [ 'Timestamp', 'Name', 'Email', 'Message', 'Source', 'Device Info' ]); const timestamp = new Date().toLocaleString('ar-EG'); const deviceInfo = data.userAgent || 'Unknown Device'; sheet.appendRow([ timestamp, data.name || '', data.email || '', data.message || '', data.source || 'website', deviceInfo ]); return createResponse('success', 'Contact saved successfully'); } function saveQuickOrder(data, spreadsheet) { let sheet = getOrCreateSheet(spreadsheet, 'Quick Orders', [ 'Timestamp', 'Product Name', 'Product Price', 'Product Category', 'Order Type', 'Source', 'Device Info' ]); const timestamp = new Date().toLocaleString('ar-EG'); const deviceInfo = data.userAgent || 'Unknown Device'; sheet.appendRow([ timestamp, data.productName || '', data.productPrice || '', data.productCategory || '', data.orderType || 'quick_order', data.source || 'website', deviceInfo ]); return createResponse('success', 'Quick order saved successfully'); } function getCompleteOrders(spreadsheet) { const sheet = spreadsheet.getSheetByName('Complete Orders'); if (!sheet) { return createResponse('success', []); } const data = sheet.getDataRange().getValues(); const orders = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (row[1]) { // Check if Order ID exists orders.push({ timestamp: row[0], orderId: row[1], customerName: row[2], customerPhone: row[3], customerEmail: row[4], customerCity: row[5], customerAddress: row[6], paymentMethod: row[7], customerNotes: row[8], products: row[9], totalItems: row[10], totalAmount: row[11] || 0, source: 'google_sheets', deviceInfo: row[13] || 'Unknown Device' }); } } return createResponse('success', orders); } function getAllData(spreadsheet) { const completeOrders = getCompleteOrders(spreadsheet); // Get contacts const contactsSheet = spreadsheet.getSheetByName('Contacts'); const contacts = []; if (contactsSheet) { const contactData = contactsSheet.getDataRange().getValues(); for (let i = 1; i < contactData.length; i++) { const row = contactData[i]; if (row[1]) { contacts.push({ timestamp: row[0], name: row[1], email: row[2], message: row[3], source: row[4] || 'website', deviceInfo: row[5] || 'Unknown Device' }); } } } // Get quick orders const quickOrdersSheet = spreadsheet.getSheetByName('Quick Orders'); const quickOrders = []; if (quickOrdersSheet) { const quickData = quickOrdersSheet.getDataRange().getValues(); for (let i = 1; i < quickData.length; i++) { const row = quickData[i]; if (row[1]) { quickOrders.push({ timestamp: row[0], productName: row[1], productPrice: row[2], productCategory: row[3], orderType: row[4] || 'quick_order', source: row[5] || 'website', deviceInfo: row[6] || 'Unknown Device' }); } } } return createResponse('success', { completeOrders: completeOrders.data || [], contacts: contacts, quickOrders: quickOrders }); } function getOrCreateSheet(spreadsheet, sheetName, headers) { let sheet = spreadsheet.getSheetByName(sheetName); if (!sheet) { sheet = spreadsheet.insertSheet(sheetName); if (headers && headers.length > 0) { sheet.getRange(1, 1, 1, headers.length).setValues([headers]); sheet.getRange(1, 1, 1, headers.length).setFontWeight('bold'); sheet.getRange(1, 1, 1, headers.length).setBackground('#f0f0f0'); } } return sheet; } function createResponse(status, data) { const response = { status: status, data: data, timestamp: new Date().toISOString() }; return ContentService .createTextOutput(JSON.stringify(response)) .setMimeType(ContentService.MimeType.JSON); }

الخطوة 3: النشر النهائي (مرة واحدة فقط)

مهم جداً: هذه الخطوة تتم مرة واحدة فقط، والرابط لن يتغير أبداً!
  1. اضغط Deploy > New deployment
  2. اختر Type: Web app
  3. Execute as: Me (your-email@gmail.com)
  4. Who has access: Anyone
  5. اضغط Deploy
  6. انسخ الرابط واحفظه للأبد!
