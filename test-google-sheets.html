إذا فشل اختبار GET، انسخ هذا الكود لـ Google Apps Script:

// Enhanced Google Apps Script for Forsa Database function doGet(e) { try { const action = e.parameter.action; const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); switch(action) { case 'getCompleteOrders': return getCompleteOrders(spreadsheet); default: return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: 'Unknown action: ' + action })).setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function getCompleteOrders(spreadsheet) { const sheet = spreadsheet.getSheetByName('Complete Orders'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ status: 'success', data: [] })).setMimeType(ContentService.MimeType.JSON); } const data = sheet.getDataRange().getValues(); const orders = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (row[1]) { orders.push({ orderId: row[1], customerName: row[2], customerPhone: row[3], totalAmount: row[11] || 0 }); } } return ContentService.createTextOutput(JSON.stringify({ status: 'success', data: orders })).setMimeType(ContentService.MimeType.JSON); } // Your existing doPost function should remain here...
