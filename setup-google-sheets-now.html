🔧 حل مشكلة: الطلبات من الأجهزة الأخرى مش بتظهر

🚨 المشكلة اللي بتواجهها:

لما العميل يعمل طلب من تليفون تاني، الطلب مش بيظهر في لوحة إدارة البيانات

ده لأن كل جهاز بيحفظ بياناته المحلية بس، والأجهزة مش مترابطة ببعض

✅ الحل:

ربط النظام بـ Google Sheets عشان جميع الطلبات من كل الأجهزة تتجمع في مكان واحد

⚡ إعداد سريع في 5 دقائق

اتبع الخطوات دي وهتحل المشكلة نهائياً

الخطوة 1: إنشاء Google Sheets

1. اذهب إلى sheets.google.com

2. اضغط "ملف جديد" أو "Blank"

3. اسم الملف "Forsa Orders Database"

الخطوة 2: إضافة الكود

1. في Google Sheets، اضغط Extensions > Apps Script

2. امسح أي كود موجود

3. انسخ والصق الكود ده:

function doPost(e) { try { const data = JSON.parse(e.postData.contents); const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'saveCompleteOrder') { return saveCompleteOrder(data, spreadsheet); } else if (data.action === 'saveContact') { return saveContact(data, spreadsheet); } else if (data.action === 'saveOrder') { return saveQuickOrder(data, spreadsheet); } return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: 'Unknown action' })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const action = e.parameter.action; const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'getCompleteOrders') { return getCompleteOrders(spreadsheet); } return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: 'Unknown action: ' + action })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function saveCompleteOrder(data, spreadsheet) { let sheet = spreadsheet.getSheetByName('Complete Orders'); if (!sheet) { sheet = spreadsheet.insertSheet('Complete Orders'); sheet.getRange(1, 1, 1, 13).setValues([[ 'Timestamp', 'Order ID', 'Customer Name', 'Customer Phone', 'Customer Email', 'Customer City', 'Customer Address', 'Payment Method', 'Customer Notes', 'Products JSON', 'Total Items', 'Total Amount', 'Source' ]]); } const timestamp = new Date().toLocaleString('ar-EG'); const products = typeof data.products === 'string' ? data.products : JSON.stringify(data.products || []); sheet.appendRow([ timestamp, data.orderId || '', data.customerName || '', data.customerPhone || '', data.customerEmail || '', data.customerCity || '', data.customerAddress || '', data.paymentMethod || '', data.customerNotes || '', products, data.totalItems || 0, data.totalAmount || 0, 'google_sheets' ]); return ContentService.createTextOutput(JSON.stringify({ status: 'success', message: 'Order saved successfully' })).setMimeType(ContentService.MimeType.JSON); } function saveContact(data, spreadsheet) { let sheet = spreadsheet.getSheetByName('Contacts'); if (!sheet) { sheet = spreadsheet.insertSheet('Contacts'); sheet.getRange(1, 1, 1, 4).setValues([['Timestamp', 'Name', 'Email', 'Message']]); } const timestamp = new Date().toLocaleString('ar-EG'); sheet.appendRow([timestamp, data.name, data.email, data.message]); return ContentService.createTextOutput(JSON.stringify({ status: 'success' })).setMimeType(ContentService.MimeType.JSON); } function saveQuickOrder(data, spreadsheet) { let sheet = spreadsheet.getSheetByName('Quick Orders'); if (!sheet) { sheet = spreadsheet.insertSheet('Quick Orders'); sheet.getRange(1, 1, 1, 5).setValues([['Timestamp', 'Product Name', 'Product Price', 'Product Category', 'Order Type']]); } const timestamp = new Date().toLocaleString('ar-EG'); sheet.appendRow([timestamp, data.productName, data.productPrice, data.productCategory, data.orderType]); return ContentService.createTextOutput(JSON.stringify({ status: 'success' })).setMimeType(ContentService.MimeType.JSON); } function getCompleteOrders(spreadsheet) { const sheet = spreadsheet.getSheetByName('Complete Orders'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ status: 'success', data: [] })).setMimeType(ContentService.MimeType.JSON); } const data = sheet.getDataRange().getValues(); const orders = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (row[1]) { orders.push({ orderId: row[1], customerName: row[2], customerPhone: row[3], customerEmail: row[4], customerCity: row[5], customerAddress: row[6], paymentMethod: row[7], customerNotes: row[8], products: row[9], totalItems: row[10], totalAmount: row[11] || 0, source: 'google_sheets', timestamp: row[0] }); } } return ContentService.createTextOutput(JSON.stringify({ status: 'success', data: orders })).setMimeType(ContentService.MimeType.JSON); }

الخطوة 3: نشر التطبيق

1. احفظ الكود: اضغط Ctrl + S

2. اضغط Deploy > New deployment

3. اختار Type: Web app

4. Execute as: Me

5. Who has access: Anyone (مهم جداً!)

6. اضغط Deploy

7. انسخ الـ URL اللي هيظهر (ينتهي بـ /exec)
